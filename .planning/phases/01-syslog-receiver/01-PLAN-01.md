#### T-03: Implement circular buffer
<read_first>
- main.go (task T-02)
</read_first>
<action>
Add to `main.go`:
```go
type LogEntry struct {
    Timestamp time.Time
    SourceIP  string
    SourcePort int
    Message   string
}

type CircularBuffer struct {
    entries [20]LogEntry
    count   int
    head    int // index of oldest
    mu      sync.Mutex
}

func (cb *CircularBuffer) Add(entry LogEntry) {
    cb.mu.Lock()
    defer cb.mu.Unlock()
    cb.entries[cb.head] = entry
    cb.head = (cb.head + 1) % 20
    if cb.count < 20 {
        cb.count++
    }
}

func (cb *CircularBuffer) GetAll() []LogEntry {
    cb.mu.Lock()
    defer cb.mu.Unlock()
    result := make([]LogEntry, 0, cb.count)
    for i := 0; i < cb.count; i++ {
        idx := (cb.head - cb.count + i + 20) % 20
        result = append(result, cb.entries[idx])
    }
    // Reverse to get newest first
    for i, j := 0, len(result)-1; i < j; i, j = i+1, j-1 {
        result[i], result[j] = result[j], result[i]
    }
    return result
}
```
</action>
<acceptance_criteria>
- `CircularBuffer` struct exists in main.go
- `Add()` appends entry and evicts oldest when at capacity
- `GetAll()` returns entries newest first
- `sync.Mutex` protects concurrent access
</acceptance_criteria>

```go
type Hub struct {
    clients    map[*websocket.Conn]bool
    broadcast  chan LogEntry
    register   chan *websocket.Conn
    unregister chan *websocket.Conn
    mu         sync.RWMutex
}

func (h *Hub) Run() {
    for {
        select {
        case client := <-h.register:
            h.mu.Lock()
            h.clients[client] = true
            h.mu.Unlock()
            // Send current buffer on connect
            for _, entry := range logBuffer.GetAll() {
                data, _ := json.Marshal(entry)
                client.WriteMessage(websocket.TextMessage, data)
            }
        case client := <-h.unregister:
            h.mu.Lock()
            delete(h.clients, client)
            h.mu.Unlock()
            client.Close()
        case entry := <-h.broadcast:
            data, _ := json.Marshal(entry)
            h.mu.RLock()
            for client := range h.clients {
                client.WriteMessage(websocket.TextMessage, data)
            }
            h.mu.RUnlock()
        }
    }
}

var hub = Hub{
    clients:    make(map[*websocket.Conn]bool),
    broadcast:  make(chan LogEntry, 100),
    register:   make(chan *websocket.Conn),
    unregister: make(chan *websocket.Conn),
}
```
